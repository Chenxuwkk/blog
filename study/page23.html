    while(left <= right)
    {
        int mid = left + right >> 1;
        if(arr[mid] < k)

        else
            return true;

{

# 两个数组的交集 II（350）

  1. 简单来说就是先对两个数组进行排序, 然后遍历第一个数组里面的元素, 二分查找另一个数组里面的元素
  2. 为了处理有多个相同元素的情况, 就要分别在两个数组中二分搜索相同元素的第一个下标和最后一个下标
  3. 比如: nums1 = [1,2,2,1], nums2 = [1,1]
  4. 排序之后为: nums1 = [1,1,2,2], nums2 = [1,1]
  5. 第一个数组中元素1的开始下标为0, 结束下标为1
  6. 第二个数组中元素1的开始下标为0, 结束下标为1
  7. 取其中最小的数作为total(因为这个例子中都是2, 所以就是2)
  8. 把当前元素1放到结果数组中, 循环total遍
  9. 然后继续遍历第一个数组中的元素, 当前遍历的位置就是 第一个数组中元素1的结束下标+1, 即 1 + 1 = 2

个人理解

本题比较需要注意的地方在于此次搜索为下次搜索排除了无用区间。

从这题开始往下就是二分的优化问题

//二分查找+缩短搜索区间
int cmpfunc (const void * a, const void * b)
{
   return ( *(int*)a - *(int*)b );
}
//左闭右开 [left, right) 
int left_bound(int* nums, int target,int numsSize) {
    if (numsSize == 0) return -1;
    int left = 0;
    int right = numsSize; // 注意
    
    while (left < right) { // 注意
        int mid = left + (right - left) / 2;
        if (nums[mid] == target) {
            right = mid;
        } else if (nums[mid] < target) {
            left = mid + 1;
        } else if (nums[mid] > target) {
            right = mid; // 注意
        }
    }
    return left;
}
int right_bound(int* nums, int target,int numsSize) {
    if (numsSize == 0) return -1;
    int left = 0, right = numsSize;
    
    while (left < right) {
        int mid = left + (right - left) / 2;
        if (nums[mid] == target) {
            left = mid + 1; // 注意
        } else if (nums[mid] < target) {
            left = mid + 1;
        } else if (nums[mid] > target) {
            right = mid;
        }
    }
    return left - 1; // 注意
}

int* intersect(int* nums1, int nums1Size, int* nums2, int nums2Size, int* returnSize){
    int* ans = calloc((nums1Size>nums2Size?nums2Size:nums1Size)+1,sizeof(int));
    qsort(nums1,nums1Size,sizeof(int),cmpfunc);
    qsort(nums2,nums2Size,sizeof(int),cmpfunc);
    
    *returnSize = 0;
    int la = 0;
    while(la<nums1Size)
    {
        int ra = right_bound(nums1,nums1[la],nums1Size);
        int lb = left_bound(nums2,nums1[la],nums2Size);
        if(lb!=-1)
        {
            int rb = right_bound(nums2,nums1[la],nums2Size);
            int totalA = ra-la+1;
            int totalB = rb-lb+1;
            int total = fmin(totalA,totalB);
            int tmp = *returnSize;
            while( *returnSize < total+tmp)
            {
                ans[(*returnSize)++] = nums1[la];
            }
        }
        la = ra+1;
    }
    return ans;
}
