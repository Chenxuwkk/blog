# C/C++（语言中连续赋值问题

错误写法：定义时连续赋值

#include<stdio.h>
int main()
{
int a=b=c=1;
printf("%d,%d,%d\n",a,b,c);
return 0;
}

正确写法

#include<stdio.h>                    
int main()
{
	int a,b,c;
	a=b=c=1;
	printf("%d %d %d\n",a,b,c);
	return 0;
}

tips

动态规划（Dynamic Programming,DP）问题的一般形式就是求最值。求解动态规划的核心问题是穷举。

  1. 列出正确的「状态转移方程」
  2. 判断算法问题是否具备「最优子结构」
  3. 动态规划问题是否存在「重叠子问题」（使用「备忘录」或者「DP table」来优化穷举过程）

# 框架

明确 base case -> 明确「状态」-> 明确「选择」 -> 定义 dp 数组/函数的含义。

# 自顶向下递归的动态规划
def dp(状态1, 状态2, ...):
    for 选择 in 所有可能的选择:
        # 此时的状态已经因为做了选择而改变
        result = 求最值(result, dp(状态1, 状态2, ...))
    return result

# 自底向上迭代的动态规划
# 初始化 base case
dp[0][0][...] = base case
# 进行状态转移
for 状态1 in 状态1的所有取值：
    for 状态2 in 状态2的所有取值：
        for ...
            dp[状态1][状态2][...] = 求最值(选择1，选择2...)

**0-1背包问题：**这个题目中的物品不可以分割，要么装进包里，要么不装，不能说切成两块装一半。这就是 0-1 背包这个名词的来历。就是「装进背包」或者「不装进背包」

# 斐波那契数（509）

极其简单

//1暴力递归
int fib(int n){
    if(n==0)return 0;
    if(n==1)return 1;
    return fib(n-1)+fib(n-2);
}

//2备忘录剪枝
int fib(int n){
    if(n==0)return 0;
    int* dp = (int*)calloc(n+1,sizeof(int));
    dp[1] = 1;
    for(int i=2;i<=n;i++)
    {
        dp[i] = dp[i-1]+dp[i-2];
    }
    return dp[n];
}

//3继续优化就是优化dp占用空间
int fib(int n) {
    if (n == 0 || n == 1) {
        return n;
    }
    // 分别代表 dp[i - 1] 和 dp[i - 2]
    int dp_i_1 = 1, dp_i_2 = 0;
    for (int i = 2; i <= n; i++) {
        int dp_i = dp_i_1 + dp_i_2;
        // 滚动更新
        dp_i_2 = dp_i_1;
        dp_i_1 = dp_i;
    }
    return dp_i_1;
}

# 第 N 个泰波那契数（1137）

暴力递归超时，用备忘录剪枝

//备忘录剪枝
int tribonacci(int n){
    if(n==0)return 0;
    if(n==1)return 1;
    if(n==2)return 1;
    int* dp = (int*)calloc(n+1,sizeof(int));
    dp[1] = 1;
    dp[2] = 1;
    for(int i=3;i<=n;i++)
    {
        dp[i] = dp[i-1]+dp[i-2]+dp[i-3];
    }
    return dp[n];
}

class Solution {
public:
    int climbStairs(int n) {
        vector<int> dp(n+2);
        dp[0] = 1;
        dp[1] = 1;
        for(int i=2;i<=n;i++)
        {
            dp[i] = dp[i-1]+dp[i-2]; 
        }
        return dp[n];
    }
};

    num_2=1;

# 使用最小花费爬楼梯（746）

int minCostClimbingStairs(int* cost, int costSize){
    int*dp = (int*)calloc(costSize+1,sizeof(int));
    for(int i=2;i<=costSize;i++)
    {
        dp[i] = fmin(dp[i-1]+cost[i-1],dp[i-2]+cost[i-2]);
    }
    return dp[costSize];
}

////优化掉dp空间
int minCostClimbingStairs(int* cost, int costSize) {
    int prev = 0, curr = 0;
    for (int i = 2; i <= costSize; i++) {
        int next = fmin(curr + cost[i - 1], prev + cost[i - 2]);
        prev = curr;
        curr = next;
    }
    return curr;
}

# 打家劫舍（198）

int rob(int* nums, int numsSize){
    if(numsSize==1)return nums[0];
    int*dp = (int*)calloc(numsSize+1,sizeof(int));
    //[0,2]一个区间
    //[2,n]一个区间
    //合起来就是[0,n]
    dp[0] = nums[0];
    dp[1] = fmax(nums[0],nums[1]);
    for(int i=2;i<numsSize;i++)
    {
        dp[i] = fmax(dp[i-2]+nums[i],dp[i-1]);
    }
    return dp[numsSize-1];
}

# 打家劫舍 II（213）

int rob(int* nums, int numsSize){
    if(numsSize==1)return nums[0];
    //[0,n-1]一个区间
    int* dp_1 = (int*)calloc(numsSize+1,sizeof(int));
    dp_1[0] = nums[0];
    dp_1[1] = fmax(nums[0],nums[1]);
    for(int i=2;i<numsSize-1;i++)
    {
        dp_1[i] = fmax(dp_1[i-2]+nums[i],dp_1[i-1]);
    }
    //[1,n]一个区间
    int* dp_2 = (int*)calloc(numsSize+1,sizeof(int));
    dp_2[0] = 0;
    dp_2[1] = nums[1];
    for(int i=2;i<numsSize;i++)
    {
        dp_2[i] = fmax(dp_2[i-2]+nums[i],dp_2[i-1]);
    }
    //最后找两个区间最大的
    return fmax(dp_1[numsSize-2],dp_2[numsSize-1]);
}

# 删除并获得点数（740）

int deleteAndEarn(int* nums, int numsSize){
    int max = -1;
    for(int i=0;i<numsSize;i++)
    {
        max = fmax(max,nums[i]);
    }
    
    int* cnt = (int*)calloc(max+1,sizeof(int));
    int* dp = (int*)calloc(max+1,sizeof(int));
    for(int i=0;i<numsSize;i++)
    {
        cnt[nums[i]]++;
    }
    //开始用cnt打家劫舍
    dp[0] = cnt[0];
    dp[1] = fmax(cnt[0],cnt[1]);
    for(int i=2;i<=max;i++)
    {
        dp[i] = fmax(dp[i-2]+i*cnt[i],dp[i-1]);
    }
    return dp[max];
}

bool canJump(int* nums, int numsSize){
    int max = 0;
    for(int i=0;i<numsSize;i++)
    {
        if(max<i)return false;
        max = fmax(max,i+nums[i]);

}

# 跳跃游戏 II（45）

int jump(int* nums, int numsSize){
    int max = 0;
    int end = 0;
    int step = 0;
    for(int i=0;i<numsSize-1;i++)
    {
        max = fmax(max,i+nums[i]);
        if(i==end)
        {
            end = max;
            step++;
        }
    }
    return step;
}

# 最大子数组和（53）

之前的题，是已经选好了dp[i-1]，然后来考虑要不要选第i个值。这道题是一定要选第i个值，然后再考虑要不要选第dp[i-1]

//这个题是画出子问题的解，然后再解中找最大
int maxSubArray(int* nums, int numsSize){
    int* dp = (int*)calloc(numsSize+1,sizeof(int));
    int max = nums[0];
    for(int i = 1;i<=numsSize;i++)
    {
        dp[i] = fmax(nums[i-1],dp[i-1]+nums[i-1]);
        max = fmax(dp[i],max);
    }
    return max;
}

# 环形子数组的最大和（918）

  1. 最大子数组不成环——53题 也就是maxSum为答案
  2. 最大子数组成环 ，那么最小子数组就不会成环——(total - minSum) 则为答案

状态转移：最大的环形子数组和 = max(最大子数组和，数组总和-最小子数组和)

int maxSubarraySumCircular(int* nums, int numsSize){
    int* dp_max = (int*)calloc(numsSize+1,sizeof(int));
    int* dp_min = (int*)calloc(numsSize+1,sizeof(int));
    int max = nums[0];
    int min = nums[0];
    int total = 0;
    //计算最大子数组和
    for(int i = 1;i<=numsSize;i++)
    {
        total+=nums[i-1];
        dp_max[i] = fmax(nums[i-1],dp_max[i-1]+nums[i-1]);
        max = fmax(dp_max[i],max);
    }
    //计算最小子数组和
    for(int i = 1;i<=numsSize;i++)
    {
        dp_min[i] = fmin(nums[i-1],dp_min[i-1]+nums[i-1]);
        min = fmin(dp_min[i],min);
    }
    return max>=0?fmax(max,total-min):max;
}

# 乘积最大子数组（152）

状态转移：

fmax⁡(i)=max⁡i=1{fmax⁡(i−1)×ai,fmin⁡(i−1)×ai,ai}fmin⁡(i)=min⁡i=1{fmax⁡(i−1)×ai,fmin⁡(i−1)×ai,ai}f_{\max}(i)=\max_{i=1}\{f_{\max}(i-1)\times a_i,f_{\min}(i-1)\times a_i,a_i\} \newline f_{\min}(i)=\min_{i=1}\{f_{\max}(i-1)\times a_i,f_{\min}(i-1)\times a_i,a_i\} fmax​(i)=i=1max​{fmax​(i−1)×ai​,fmin​(i−1)×ai​,ai​}fmin​(i)=i=1min​{fmax​(i−1)×ai​,fmin​(i−1)×ai​,ai​}

由于存在负数，那么会导致最大的变最小的，最小的变最大的。因此还需要维护当前最小值。

int maxProduct(int* nums, int numsSize){
    int* dp_max = (int*)calloc(numsSize+1,sizeof(int));
    int* dp_min = (int*)calloc(numsSize+1,sizeof(int));
    dp_max[0]=1;
    dp_min[0]=1;
    int max = nums[0];
    for(int i = 1;i<=numsSize;i++)
    {
        dp_max[i] = fmax(dp_max[i-1]*nums[i-1],fmax(dp_min[i-1]*nums[i-1],nums[i-1]));
        dp_min[i] = fmin(dp_min[i-1]*nums[i-1],fmin(dp_max[i-1]*nums[i-1],nums[i-1]));
        max = fmax(dp_max[i],max);
    }
    return max;
}

class Solution {
public:
    int maxProduct(vector<int>& nums) {
        int n = nums.size();
        vector<vector<int>> dp(n+1,vector<int>(2,0));
        dp[0][0] = 1;//维护最大值
        dp[0][1] = 1;//维护最小值
        int Max = nums[0];
        for(int i=1;i<=n;i++)
        {
            //三个
            dp[i][0] = max(dp[i-1][0]*nums[i-1],max(dp[i-1][1]*nums[i-1],nums[i-1]));
            dp[i][1] = min(dp[i-1][0]*nums[i-1],min(dp[i-1][1]*nums[i-1],nums[i-1]));
            Max = max(Max,dp[i][0]);
        }
        return Max;
    }
};

# 乘积为正数的最长子数组长度（1567）

两个dp数组轮流交替的使用。

int getMaxLen(int* nums, int numsSize){
    int* dpZ = (int*)calloc(numsSize+1,sizeof(int));
    int* dpF = (int*)calloc(numsSize+1,sizeof(int));
    if(nums[0]>0)dpZ[0]=1;
    else if(nums[0]<0)dpF[0]=1;
    int max = dpZ[0];
    for(int i=1;i<numsSize;i++)
    {
        if(nums[i]>0)
        {
            dpZ[i] = dpZ[i-1]+1;
            dpF[i] = dpF[i-1]==0?0:dpF[i-1]+1;
        }else if(nums[i]<0)
        {
            dpZ[i] = dpF[i-1]==0?0:dpF[i-1]+1;
            dpF[i] = dpZ[i-1]+1;
        }else
        {
            dpZ[i]=0;
            dpF[i]=0;
        }
        max = fmax(max,dpZ[i]);
    }
    return max;
}

# 最佳观光组合（1014）

int maxScoreSightseeingPair(int* values, int valuesSize){
    int* dp = (int*)calloc(valuesSize+1,sizeof(int));
    //dp数组保存当前最大的values
    //景点每往后移动一位价值-1
    dp[0] = values[0];
    int max = 0;
    for(int i=1;i<valuesSize;i++)
    {
        dp[i] = fmax(dp[i-1]-1,values[i]);
        max = fmax(max,dp[i-1]-1+values[i]);
    }
    return max;
}

# 买卖股票的最佳时机（121）

模板：https://labuladong.github.io/algo/di-er-zhan-a01c6/yong-dong--63ceb/yi-ge-fang-3b01b/

int maxProfit(int* prices, int pricesSize){
    int** dp = (int**)malloc(sizeof(int*)*(pricesSize+1));
    for(int i=0;i<=pricesSize;i++)
    {
        dp[i] = (int*)calloc(2,sizeof(int));
    }
    for(int i=0;i<pricesSize;i++)
    {
        if (i - 1 == -1) {
            // base case 1
            dp[i][0] = 0;
            dp[i][1] = -prices[i];
            continue;
        }
        dp[i][0] = fmax(dp[i-1][0],dp[i-1][1]+prices[i]);
        dp[i][1] = fmax(dp[i-1][1],-prices[i]);
    }
    return dp[pricesSize-1][0];
}

    int ans = 0;
    leftMax[0]=height[0];
    for(int i=1;i<heightSize;i++)
    {
        leftMax[i] = fmax(leftMax[i-1],height[i]);
    }
    rightMax[heightSize-1] = height[heightSize-1];
    for(int i=heightSize-2;i>=0;i--)
    {
        rightMax[i] = fmax(rightMax[i+1],height[i]);
    }

    }

}

    {

        dp[i] = (int*)calloc(matColSize[0]+1,sizeof(int));
    }
    (*returnSize)=matSize;

    }

    {
        for(int j=1;j<=matColSize[0];j++)
        {

        }
    }
    //dp开始
    for(int i=0;i<matSize;i++)
    {
        for(int j=0;j<matColSize[0];j++)
        {

            if(j-k<0)y1 = 1;
            if(i+k>matSize-1)x2 = matSize;
            if(j+k>matColSize[0]-1)y2 = matColSize[0];

        }
    }

char * longestPalindrome(char * s){
    int len = strlen(s);
    if (len < 2) 
    {
        return s;
    }
    int** dp = (int**)malloc(sizeof(int*)*(len+1));
    int maxLen = 1;
    int beginIndex = 0;
    for(int i= 0;i<len+1;i++)
        dp[i] = (int*)calloc(len+1,sizeof(int));
    
    for(int i=len-1;i>=0;i--)
    {
        for(int j=i;j<len;j++)
        {
            // 首先判断s[i]==s[j]
            if(s[i]==s[j])
            {
                //判断是否长度小于等于2
                if(j-i<=1)
                {
                    dp[i][j] = 1;//对角线都是1，因为自己和自己是回文串，还有一种可能是（bb）双数
                }else
                {
                    dp[i][j] = dp[i+1][j-1];//子串dp[i+1][j-1]如果是那dp[i][j]也是
                }
            }
            if(dp[i][j]&&j-i+1>maxLen)
            {
                maxLen = j-i+1;
                beginIndex = i;
            }
        }
    }

    char* res = calloc(maxLen+1,sizeof(char));
    s[beginIndex+maxLen] = '\0';
    strcpy(res,s+beginIndex);
    return res;
}

int longestCommonSubsequence(char * text1, char * text2){
    int m = strlen(text1), n = strlen(text2);
    int dp[m + 1][n + 1];
    memset(dp, 0, sizeof(dp));
    for(int i=1;i<=m;i++)
    {
        for(int j=1;j<=n;j++)
        {
            if(text1[i-1]==text2[j-1])
            {

            }else
            {
                //当前字符已经不相等的情况下，就应该把当前的字符也放入到之前的比较中，那么一定有dp[i][j-1]和dp[i-1][j]>=dp[i][j]。

                dp[i][j] = fmax(dp[i-1][j],dp[i][j-1]);
            }
        }
    }

class Solution {

            {

                dp[i][1] = dp[i-1][1];
            }
        }

    }

# 判断子序列（392）

//dp[i][j] 表示，s[0,j] 是否是t[0,i]的子序列
bool isSubsequence(char * s, char * t){
    int m = strlen(s);
    int n = strlen(t);
    int dp[n+1][m+1];
    //base
    memset(dp,0,sizeof(dp));
    for(int i=0;i<=n;i++)
    {
        dp[i][0] = 1;
    }

    for(int i=1;i<=n;i++)
    {
        for(int j=1;j<=m;j++)
        {
            if(i<j)continue;//剪枝
            if(s[j-1]==t[i-1])
            {
                dp[i][j] = dp[i-1][j-1];
            }else
            {
                dp[i][j] = dp[i-1][j];
            }
        }
    }
    return dp[n][m];
}

# 编辑距离（72）

//插入就是j-1，j直接往后移动
//删除就是i-1，i直接往后移动

int minDistance(char * word1, char * word2){
    int len1 = strlen(word1);
    int len2 = strlen(word2);
    int dp[len1+1][len2+1];
    memset(dp,0,sizeof(dp));
    dp[0][0] = 0;
    for(int i=1;i<=len1;i++)
    {
        dp[i][0] = dp[i-1][0]+1;
    }
    for(int i=1;i<=len2;i++)
    {
        dp[0][i] = dp[0][i-1]+1;
    }
    for(int i=1;i<=len1;i++)
    {
        for(int j=1;j<=len2;j++)
        {
            if(word1[i-1]==word2[j-1])
            {
                dp[i][j] = dp[i-1][j-1];
            }else
            {
                dp[i][j] = fmin(dp[i][j-1],fmin(dp[i-1][j],dp[i-1][j-1]))+1;
            }
        }
    }
    return dp[len1][len2];
}

# 整数拆分（343）

dp[i]表示将正整数 i拆分成至少两个正整数的和之后，这些正整数的最大乘积。

  1. 将 i 拆分成 j 和 i−j 的和，且i−j不再拆分成多个正整数，此时的乘积是 j*(i-j)
  2. 将 i 拆分成 j 和 i−j 的和，且i−j继续拆分成多个正整数，此时的乘积是 j*dp[i-j]
class Solution {
public:
    
    int integerBreak(int n) {
        vector<int> dp(n+1,0);
        dp[1] = 1;
        for(int i=2;i<=n;i++)
        {
            for(int j=1;j<i;j++)
            {
                dp[i] = max(max(dp[i],j*(i-j)),j*dp[i-j]);
            }
        }
        return dp[n];
    }
};

# 完全平方数（279）

class Solution {
public:
    int numSquares(int n) {
        // 定义：和为 i 的平方数的最小数量是 dp[i]
        vector<int> dp(n+1,0);// base case
        for(int i=1;i<=n;i++)
        {
            dp[i] = i;// 最坏的情况就是每个数都是 1，最多也就 i 个数
            for(int j=1;j*j<=i;j++)
            {   // i - j * j 只要再加一个平方数 j * j 即可凑出 i
                dp[i] = min(dp[i],dp[i-j*j]+1);
            }
        }
        return dp[n];
    }
};

class Solution {

# 经过一次操作后的最大子数组和（1764）

class Solution {
public:
    int maxSumAfterOperation(vector<int>& nums) {
        int n = nums.size();
        vector<vector<int>> dp(n+1,vector<int>(2,0));
        dp[0][0] = nums[0];//没有操作过
        dp[0][1] = nums[0]*nums[0];//操作过1次
        int ans = dp[0][1];
        for(int i=1;i<n;i++)
        {
            dp[i][0] = max(dp[i-1][0],0)+nums[i];
            dp[i][1] = max(max(dp[i-1][0],0)+nums[i]*nums[i],dp[i-1][1]+nums[i]);
            ans = max(ans,dp[i][1]);
        }
        return ans;
    }
};

# 抛掷硬币（1230）

class Solution {
public:
    double probabilityOfHeads(vector<double>& prob, int target) {
        int n = prob.size();
        vector<vector<double>> dp(n+1,vector<double>(target+1,0));
        dp[0][0]=1;
        for(int i=1;i<=n;i++)
        {
            dp[i][0] = dp[i-1][0]*(1-prob[i-1]);
        }
        for(int i=1;i<=n;i++)
        {
            for(int j=1;j<=target;j++)
            {
                //使用i-1个硬币凑成j-1个正面，且第i个硬币为正面
                dp[i][j]+=dp[i-1][j-1]*prob[i-1];
                //使用i-1个硬币凑成j个正面，且第i个硬币为反面
                dp[i][j]+=dp[i-1][j]*(1-prob[i-1]);
            }
        }
        return dp[n][target];
    }
};
