# 字符串操作函数

以str开头的函数 都是字符串操作函数 都是遇到'\0'结束操作

  1. 字符串拷贝函数strcpy strncpy
#include <string.h>
char *strcpy(char *dest, const char *src);
char *strncpy(char *dest, const char *src, size_t n);
//dest:目的空间地址
//src：原字符串的首元素地址

void test()
{
    char dst[128] = "";

    strcpy(dst, src);

    strcpy(dst1, src1); //dst1只有1字节 拷贝越界

  1. 字符串比较strcmp
#include <string.h>
int strcmp(const char *s1, const char *s2);
int strncmp(const char *s1, const char *s2, size_t n);

返回值：

>0 s1字符串 > s2字符串

<0 s1字符串 < s2字符串

==0 s1字符串 ==s2字符串

void test()
{
    char s1[128] = "";
    char s2[128] = "";

    printf("请输入第一个字符串:");
    scanf("%s", s1);
    printf("请输入第二个字符串:");
    scanf("%s", s2);

    if (strcmp(s1, s2) > 0) 
    {
        printf("%s 大于 %s\n", s1, s2);
    }
    else if (strcmp(s1, s2) < 0)
    {
        printf("%s 小于 %s\n", s1, s2);
    }
    else if (strcmp(s1, s2) == 0)
    {
        printf("%s 等于 %s\n", s1, s2);
    }
}

  1. 字符串 切割strtok
#include <string.h>
char *strtok(char *str, const char *delim);

第一次切割：str必须指向 待切割的字符串的首元素地址 delim指向分割符":"

后续切割：str传NULL delim指向分割符":"

返回值:

成功：返回值子串的首元素地址

失败：返回NULL

# 案例1

//hehehe:xixixi:hahaha:lalala:heiheihei:henhenhen:wuwuwu
void test()
{
    char str[] = "hehehe:xixixi:hahaha:lalala:heiheihei:henhenhen:wuwuwu";
    char *buf[32]; //存放子串的首元素地址
    //第一次切
    int i = 0;
    buf[i] = strtok(str, ":");
    //后续切：上一次切割正常
    while (buf[i] != NULL)
    {
        i++;
        buf[i] = strtok(NULL, ":");
    }

    //遍历切割到的子串
    i = 0;
    while (buf[i] != NULL)
    {
        printf("%s\n", buf[i]);
        i++;
    }
}

# 案例2

# 将字符串 转成 数值

int my_atoi(char *str)
{
    int sum = 0;
    while (*str >= '0' && *str <= '9')
    {
        sum = sum * 10 + (*str - '0');
        str++;
    }

    return sum;
}
void test()
{
    char buf[] = "12345abc";

    printf("%d\n", my_atoi(buf));
}

# 格式化字符串

组包：按照需要的格式 组成字符串 解包：解析特定格式的数据

# sprintf 用于组包

将零散的数据 按照固定的格式 组成字符串

#include <stdio.h>
int sprintf(char *str, const char *format,...);
//*str存放组包后结果
//sprintf返回值为实际组包的长度

#include <stdio.h>
void test()
{
    int year = 2021;
    int month = 7;
    int day = 30;

    char buf[128] = "";
    int len = sprintf(buf, "%d年%d月%d日", year, month, day);
    printf("len=%d, buf=%s\n", len, buf);
}
int main(int argc, char const *argv[])
{
    test();
    return 0;
}

# sscanf 用于解包

void test()
{

# %[a-z] 表示匹配 a 到 z 中任意字符(尽可能多的匹配)

void test()

}

void test()
{

# %[aBc] 匹配 a、B、c 中一员，贪婪性

void test()
{

# 综合案例

分割歌词

void test()
{
    char buf[128] = "[02:04.94][00:36.09]我想大声宣布 对你依依不舍";

    char *song_lrc = buf;
    //定位到歌词的位置
    while (*song_lrc == '[')
    {
        song_lrc += 10;
    }

    //逐个时间分析
    char *time_lrc = buf;
    while (*time_lrc == '[')
    {
        int m = 0;
        int s = 0;
        sscanf(time_lrc, "[%d:%d", &m, &s);
        printf("时间%d秒 唱歌词:%s\n", m * 60 + s, song_lrc);

        time_lrc += 10;
    }
}

提取短信

#include <string.h>
void test()
{
    char buf[] = "+CMGR:REC UNREAD,+8613466630259,98/10/01,18:22:11+00,ABCdefGHI";
    char *msg[32] = {buf};//接收切割字符串

    int i = 0;
    while ((msg[i] = strtok(msg[i], ",")) && (++i));//切割字符串

    //短信的读取状态
    //msg[0] = "+CMGR:REC UNREAD"
    char status[128] = "";
    sscanf(msg[0], "%*s %s", status);
    if (strcmp(status, "UNREAD") == 0)
    {
        printf("有未读信息\n");
    }
    else if (strcmp(status, "READ") == 0)
    {
        printf("已读信息\n");
    }

    //msg[1]="+8613466630259"
    printf("手机号:%s\n", msg[1] + 3);

    //msg[2]="98/10/01"
    int year = 0;
    int month = 0;
    int day = 0;
    sscanf(msg[2], "%d/%d/%d", &year, &month, &day);
    printf("日期:%02d年%02d月%02d日\n", year + 1900, month, day);

    //msg[3]="18:22:11+00"
    int h = 0, m = 0, s = 0;
    sscanf(msg[3], "%d:%d:%d", &h, &m, &s);
    printf("时间:%02d时%02d分%02d秒\n", h, m, s);

    printf("收到的消息:%s\n", msg[4]);
}

# const修饰

# const修饰普通变量 为只读变量

void test()
{
    //num为只读  只能初始化 不能被赋值
    const int num = 10;
    // num = 100; //err
    printf("num = %d\n", num);
}

# const 修饰 *

const int *p
//在使用中：
//*p是只读 不同通过*p 修改p所指向的空间内容
//p 可读可写 p可以指向其他空间

void test()
{
    int num = 10;
    //p可读可写  *p只读
    const int *p = &num;
    //*p = 100;//error

    int data = 20;
    p = &data; 
    printf("*p = %d\n", *p);
}

# const 修饰 指针变量

int * const p = &num;
//p只读 除了初始化 不能修改p的指向
//*p可读可写 可以通过*p修改p指向的空间内容

void test()
{
    int num = 10;
    //p只读  *p可读可写
    int *const p = &num;
    *p = 100;

    int data = 20;
    //p = &data; //error
    printf("*p = %d\n", *p);
}

# const 既修饰 * 也修饰指针变量

const int *const p

void test()
{
    int num = 10;
    //p只读  *p只读
    const int *const p = &num;
    //*p = 100;//error

    int data = 20;
    //p = &data; //error
    printf("*p = %d\n", *p);
}
